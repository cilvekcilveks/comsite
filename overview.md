        'capability_type'     => array('kultura', 'kulturas'), // Custom caps

        'has_archive'         => true,
        'archive_slug'        => 'kultura',
        'delete_with_user'    => true,                    // Delete posts if user deleted
    );

    register_post_type('kultura', $args);
}
add_action('init', 'komunitate_register_kultura_cpt');


// Map meta capabilities for granular control
function komunitate_map_meta_caps($caps, $cap, $user_id, $args) {
    if ('delete_kultura' === $cap) {
        // Only allow deletion if user is moderator or manager or owns it and is creator
        $post = get_post($args[0]);
        if ($post && $user_id === (int) $post->post_author) {
            // Owner can only delete if they're at least a creator
            if (user_can($user_id, 'manage_kultura_cap')) {
                return array('manage_kultura_cap');
            }
        }
        // Only moderators and above can delete others' posts
        if (user_can($user_id, 'delete_others_posts')) {
            return array('delete_others_posts');
        }
    }
    return $caps;
}
add_filter('map_meta_cap', 'komunitate_map_meta_caps', 10, 4);

// Hook into user capability checks
function komunitate_restrict_kultura_access($allcaps, $cap, $args, $user) {
    
    // Get user role
    $user_roles = $user->roles;
    
    // GUEST/ANONYMOUS: Only read
    if (empty($user_roles)) {
        if ('read' === $cap) {
            $allcaps['read'] = true;
        }
        return $allcaps;
    }

    // SUBSCRIBER/MEMBER: Read + Comment
    if (in_array('subscriber', $user_roles) || in_array('komunitate_member', $user_roles)) {
        $allcaps['read'] = true;
        $allcaps['read_kultura'] = true;
        return $allcaps;
    }

    // CONTRIBUTOR (NOT RECOMMENDED - Skip)

    // AUTHOR / CONTENT CREATOR: Create, edit own, publish
    if (in_array('author', $user_roles) || in_array('kultura_creator', $user_roles)) {
        $allcaps['read'] = true;
        $allcaps['read_kultura'] = true;
        $allcaps['create_kultura'] = true;
        $allcaps['edit_kultura'] = true;
        $allcaps['edit_published_kultura'] = true;
        $allcaps['publish_kultura'] = true;
        
        // Can only delete if creator AND post is in specific status
        if (isset($args[0])) {
            $post = get_post($args[0]);
            if ($post && $user->ID === $post->post_author) {
                $allcaps['delete_kultura'] = true;
            }
        }
        return $allcaps;
    }

    // MODERATOR: Full edit/delete but NO user management
    if (in_array('kultura_moderator', $user_roles)) {
        $allcaps['read'] = true;
        $allcaps['read_kultura'] = true;
        $allcaps['read_private_kultura'] = true;
        $allcaps['edit_kultura'] = true;
        $allcaps['edit_others_kultura'] = true;
        $allcaps['edit_published_kultura'] = true;
        $allcaps['delete_kultura'] = true;
        $allcaps['delete_others_kultura'] = true;
        $allcaps['delete_published_kultura'] = true;
        $allcaps['manage_kultura_taxonomy'] = true;
        $allcaps['moderate_comments'] = true;
        // NO: manage_users, manage_options
        return $allcaps;
    }

    // MANAGER / EDITOR: Full access
    if (in_array('editor', $user_roles) || in_array('kultura_manager', $user_roles)) {
        $allcaps['manage_options'] = true;
        $allcaps['manage_users'] = true;
        $allcaps['create_users'] = true;
        $allcaps['edit_users'] = true;
        $allcaps['delete_users'] = true;
        $allcaps['read'] = true;
        $allcaps['read_kultura'] = true;
        $allcaps['read_private_kultura'] = true;
        $allcaps['create_kultura'] = true;
        $allcaps['edit_kultura'] = true;
        $allcaps['edit_others_kultura'] = true;
        $allcaps['edit_published_kultura'] = true;
        $allcaps['publish_kultura'] = true;
        $allcaps['delete_kultura'] = true;
        $allcaps['delete_others_kultura'] = true;
        $allcaps['delete_published_kultura'] = true;
        $allcaps['manage_kultura_taxonomy'] = true;
        $allcaps['moderate_comments'] = true;
        return $allcaps;
    }

    // ADMINISTRATOR: Already has everything
    return $allcaps;
}

// Add post to user bookmarks (using user meta)
function komunitate_add_to_bookmarks($post_id, $user_id = null) {
    if (!$user_id) {
        $user_id = get_current_user_id();
    }

    if (!$user_id) return false;

    // Get existing bookmarks
    $bookmarks = get_user_meta($user_id, 'kultura_bookmarks', true);
    $bookmarks = $bookmarks ? (array) $bookmarks : array();

    // Add new bookmark if not already there
    if (!in_array($post_id, $bookmarks)) {
        $bookmarks[] = (int) $post_id;
        update_user_meta($user_id, 'kultura_bookmarks', $bookmarks);
        return true;
    }

    return false;
}

// Remove from bookmarks
function komunitate_remove_from_bookmarks($post_id, $user_id = null) {
    if (!$user_id) {
        $user_id = get_current_user_id();
    }

    if (!$user_id) return false;

    $bookmarks = get_user_meta($user_id, 'kultura_bookmarks', true);
    $bookmarks = $bookmarks ? (array) $bookmarks : array();

    // Remove bookmark
    $bookmarks = array_filter($bookmarks, function($id) use ($post_id) {
        return $id !== (int) $post_id;
    });

    update_user_meta($user_id, 'kultura_bookmarks', $bookmarks);
    return true;
}

// Get user bookmarks
function komunitate_get_user_bookmarks($user_id = null) {
    if (!$user_id) {

    if (!$user_id) return array();

    if (!$bookmarks) return array();

    // Query bookmarked posts

        'post_type'  => 'kultura',

        'orderby'    => 'post__in',

// Handle taxonomy filters
function komunitate_parse_query_kultura($query) {
    global $pagenow, $typenow;
    
    if (is_admin() && 'edit.php' === $pagenow && 'kultura' === $typenow) {
        $taxonomies = array('kultura_kategorija', 'kultura_zanrs');
        
        foreach ($taxonomies as $taxonomy) {
            if (isset($_GET[$taxonomy]) && !empty($_GET[$taxonomy])) {
                $tax_query = array(
                    array(
                        'taxonomy' => $taxonomy,
                        'field'    => 'slug',
                        'terms'    => sanitize_text_field($_GET[$taxonomy]),
                    ),
                );
                
                // Merge with existing tax_query
                if ($query->get('tax_query')) {
                    $existing = $query->get('tax_query');
                    $existing[] = $tax_query[0];
                    $query->set('tax_query', $existing);
                } else {
                    $query->set('tax_query', $tax_query);
                }
            }
        }
    }
}

## ✅ THE CORRECT ARCHITECTURE

### **Best Practice: 1 CPT + Multiple Taxonomies**

```php
// Register ONE "kultūra" CPT
function komunitate_register_kultura_cpt() {
    register_post_type('kultura', array(
        'labels' => array(
            'name' => 'Kultūra',
            'singular_name' => 'Kultūras Post'
        ),
        'public' => true,
        'has_archive' => true,
        'rewrite' => array('slug' => 'kultura'),
        'supports' => array('title', 'editor', 'author', 'thumbnail', 'excerpt', 'comments'),
        'capability_type' => array('kultura', 'kulturas'),
        'map_meta_cap' => true,
    ));
}

// Register TAXONOMIES to organize posts
function komunitate_register_kultura_taxonomies() {
    
    // PRIMARY: Jomas (Domains/Sections)
    register_taxonomy('kultura_joma', 'kultura', array(
        'label' => 'Joma',
        'hierarchical' => true,  // Can have sub-categories
        'show_ui' => true,
        'rewrite' => array('slug' => 'joma'),
        'show_in_rest' => true,
    ));
    
    // SECONDARY: Veids (Type/Genre)
    register_taxonomy('kultura_veids', 'kultura', array(
        'label' => 'Veids',
        'hierarchical' => false,  // Just tags
        'show_ui' => true,
        'rewrite' => array('slug' => 'veids'),
        'show_in_rest' => true,
    ));
}
```
